func main() {
	// 查看文件信息
	fileInfo, err := os.Stat("./data.txt")
	if err != nil {
		fmt.Println(err)
		return
	}
	fmt.Println("Stat:", fileInfo.Name())
	fmt.Println("Stat:", fileInfo.IsDir())
	fmt.Println("Stat:", fileInfo.Size())
	fmt.Println("Stat:", fileInfo.Mode())
	fmt.Println("Stat:", fileInfo.ModTime())
}

func main() {

	fmt.Println("IsAbs:", filepath.IsAbs(filename1))

	// 获取绝对路径
	fileabs, err := filepath.Abs(filename1)
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(fileabs)

	// 绝对路径的父级目录
	fmt.Println("Join:", path.Join(fileabs, ".."))
}

func main() {
    file, err := os.Open("./data.txt")
    if err != nil {
        log.Fatal(err)
    }
    defer file.Close()
 
    b := make([]byte, 128)
    for {
        n, err := file.Read(b)
        if err == io.EOF {
            break
        }
        if err != nil {
            log.Fatal(err)
        }
        fmt.Printf("%s", string(b[:n]))
    }
}

    }

}

func main() {
    file, err := os.Open("./data.txt")
    if err != nil {
        log.Fatal(err)
    }
    defer file.Close()

    reader := bufio.NewReader(file)
    for {
        str, err := reader.ReadString('\n')
        if err == io.EOF {
            break
        }
        if err != nil {
            log.Fatal(err)
        }
        fmt.Printf("%s", str)
    }
}

func main() {

    if err != nil {
        log.Fatal(err)
    }
    defer file.Close()
    
    scanner := bufio.NewScanner(file)
    for scanner.Scan() {
        fmt.Println(scanner.Text())
    }
    err = scanner.Err()
    if err != nil {
        log.Fatal(err)
    }
}
